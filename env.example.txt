# 프로덕션 배포 시에는 Vercel 대시보드에서 환경 변수 설정

# ========================================
# Supabase 설정
# ========================================
# Supabase 프로젝트에서 가져오기:
# 1. supabase.com → 프로젝트 선택
# 2. Settings → API
# 3. Project URL과 anon public 키 복사

VITE_SUPABASE_URL=https://your-project.supabase.co
VITE_SUPABASE_ANON_KEY=your-anon-public-key-here

# ========================================
# 로컬 개발 환경
# ========================================
# .env 파일 생성 후 위의 Supabase 값을 입력하세요

# ========================================
# 프로덕션 환경 (Vercel)
# ========================================
# Vercel CLI로 환경 변수 추가:
# vercel env add VITE_SUPABASE_URL production
# vercel env add VITE_SUPABASE_ANON_KEY production

# 또는 Vercel 대시보드에서:
# Project → Settings → Environment Variables

# ========================================
# 사용 방법
# ========================================
# 1. 이 파일을 .env로 복사: cp env.example.txt .env
# 2. .env 파일에서 실제 Supabase 값으로 수정
# 3. Git에 .env 파일은 커밋하지 마세요!
